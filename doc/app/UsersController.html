def create
  @user = User.new(user_params)
  if @user.save
    UserMailer.welcome_email(@user).deliver
    user = login(params[:user][:email], params[:user][:password])
    redirect_to projects_url, :notice => "Signed up!"
  else
    render :new
  end
end
